If-else-if conditions

#include <stdio.h>
int main() {
    int a;
    printf("enter value of a: ");
    scanf("%d",&a);
    if(a>0)
    {
       printf("a is positive number");
    }
    else if(a<0)
    {
        printf("a is negative number");
    }
    else
    {
        printf("a is zero");
    }
    return 0;
}
