if -else condition

#include <stdio.h>
int main() {
    int a,b;
    printf("enter value of a and b :");
    scanf("%d %d",&a,&b);
    if(a==b)
    {
        printf("a is equal to b");
    }
    else
    {
        printf("a is not equal to b");
    }
    return 0;
}

Nested-if condition

#include <stdio.h>
int main() {
    int a,b,c;
    printf("enter value of a,b and c :");
    scanf("%d %d %d",&a,&b,&c);
    if(a>b)
    {
        if(a>c)
        printf("a is greatest");
        else
        printf("c is greatest");
    }
    else
    {
        if(b>c)
        printf("b is greatest");
        else
        printf("c is greatest");
    }
    return 0;
}
