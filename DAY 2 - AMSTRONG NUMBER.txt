AMSTRONG NUMBER 


#include <stdio.h>

int main() {
   int num,ogn,r,n=0,temp,result=0;
   printf("Enter an integer : ");
   scanf("%d",&num);
   ogn=num;
   temp=num;
   while(temp!=0)
   {
       temp /= 10;
       n++;
   }
   temp=num;
   while(temp!=0)
   {
       r=temp%10;
       int p=1;
       for(int i=0;i<n;i++)
       {
           p *= r;
       }
       result += p;
       temp /= 10;
   }
   if(result == num)
   printf("%d is an Amstrong number\n",num);
   else
   printf("%d is not an Amstrong number\n",num);
    return 0;
}
