#include <stdio.h>

int main()
{
    int arr[3][3];
    int i,j;
    printf("Enter 9 numbers : ");
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            scanf("%d",&arr[i][j]);
        }
    }
    printf("Entered numbers are : \n");
   for(i=0;i<3;i++)

            printf("%d ",arr[i][j]);

    }

}
