DAY 5 - copy one to another array

#include <stdio.h>
int main()
{
   int i,n;
   printf("Enter size of an array : ");
   scanf("%d",&n);
   int a[n],b[n];
   printf("Enter %d numbers to add in an array : ",n);
   for(i=0;i<n;i++)
   {
       scanf("%d",&a[i]);
   }
   printf("A = ");
   for(i=0;i<n;i++)
   {
       printf("%d ",a[i]);
   }
   printf("\nB = ");
   for(i=0;i<n;i++)
   {
       b[i]=a[i];
       printf("%d ",b[i]);
   }
   return 0;
}
