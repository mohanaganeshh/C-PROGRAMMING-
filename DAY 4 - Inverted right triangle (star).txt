DAY 4 - Inverted right triangle (star)

#include <stdio.h>

int main()
{
    int i,n,j;

    printf("Enter a number of stars : ");

    scanf("%d",&n);

    for(i=n;i<=n;i--)
    {
        for(j=1;j<=i;j++)
        {
            printf("* ");
        }
        printf("\n");
    }
    return 0;
}
