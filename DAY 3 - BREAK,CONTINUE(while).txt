DAY 3 - BREAK,CONTINUE(while)

#include <stdio.h>
int main() {
   int i=0,n;
   printf("Enter limit : ");
   scanf("%d",&n);
    while(i<=n)
    {
        i++;
        if(i==5)
        continue;
        else if(i==9)
        break;
        else
        printf("%d\n",i);
       
    }

    return 0;
}
