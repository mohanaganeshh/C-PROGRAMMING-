Arithmetic Operators

#include <stdio.h>
int main() {
    int a,b;
    printf("enter value of a and b :");
    scanf("%d %d",&a,&b);
    printf("sum of a and b is %d\n",a+b);
    printf("difference of a and b is %d\n",a-b);
    printf("multiply of a and b is %d\n",a*b);
    printf("division of a and b is %d\n",a/b);
    return 0;
}
