DAY 5 - min numbers in array

#include <stdio.h>
int main()
{
   int i,min,a[5];
   printf("Enter 5 numbers : ");
   for(i=0;i<5;i++)
   {
       scanf("%d",&a[i]);
   }
min=a[0];
for(i=0;i<5;i++)
{
    if(a[i]<min)
       {
           min=a[i];
       }
}
   printf("Min = %d",min);
    return 0;
}
