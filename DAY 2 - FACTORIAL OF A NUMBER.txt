FACTORIAL OF A NUMBER:



#include <stdio.h>

int main() {
    int i,n,fact=1;
    printf("Enter a number : ");
    scanf("%d",&n);
    if(n<0)
    {
        printf("Factorial is not for -ve numbers\n");
    }
    else
    {
        for(i=1;i<=n;i++)
        {
            fact *= i;
        }
        printf("Factorial of %d = %d\n",n,fact);
    }
    return 0;
}
