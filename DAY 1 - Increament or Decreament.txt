DAY 1 - Increament or Decreament

Increment or Decrement 

#include <stdio.h>
int main() {
    int a,b;
    printf("enter value of a and b : ");
    scanf("%d %d",&a,&b);
    a++;
    printf("the increament of a is %d\n",a);
    b--;
    printf("the decreament of a is %d",b);
    return 0;
}
