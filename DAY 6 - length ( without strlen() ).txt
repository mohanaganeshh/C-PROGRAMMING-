DAY 6 - length ( without strlen() )

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i, l=0;

    printf("Enter a string: ");
    scanf("%s", str);

    for(i=0; str[i] != '\0';i++) {
        l++;
    }

    printf("Length of a string is : %d\n", l);

    return 0;
}
