DAY 6 - remove spaces from a string

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i, l;

    printf("Enter a string : ");
    scanf("%[^\n]",str);
    l = strlen(str);
    for(i=0;i<l;i++)
    {
        if(str[i] == ' ')
        continue;
        else
        printf("%c",str[i]);
    }

    return 0;
}
