DAY 2 - PRIME NUMBER OR NOT



#include<stdio.h>
int main()
{ 
    int n , i , a=0;
    printf("Enter a postive number:");
    scanf("%d",&n);
    if(n<=1)
    {
        printf("%d is not a prime number \n",n); 
        return 0;
    } 
    for (i=2;i<=n/2;i++)
    { 
        if (n%i== 0)
        { 
            a=1;
            break;
        }
        
    }
    if (a==0)
    { 
        printf("%d is a prime number \n",n); 
        
    } 
    else
    { 
        printf("%d is not a prime number \n",n); 
        
    } 
    return 0; 
}
